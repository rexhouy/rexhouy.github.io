Screen size

Bootstrap compatible 

@media (max-width: 768px) {
    // Small screen
}
@media (min-width:768px) and (max-width: 1200px) {
    // Middle screen
}
@media (min-width: 1200px) {
    // Large screen
}

Display Elements in different screen size

.large-screen, .middle-screen, .small-screen, .not-large-screen, .not-middle-screen, .not-small-screen {
    display: none;
}
@media (max-width: 768px) {
    .small-screen, .not-large-screen, .not-middle-screen {
        display: block;
    }
}
@media (min-width:768px) and (max-width: 1200px) {
    .middle-screen, .not-small-screen, .not-large-screen {
        display: block;
    }
}
@media (min-width: 1200px) {
    .large-screen, .not-small-screen, .not-middle-screen {
        display: block;
    }
}
