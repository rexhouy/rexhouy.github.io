查询接口filter规则

filter: {
    groupOp : 'AND', // AND | OR
    rules: [{
        field: 'field id',
        op: 'in',
        /*op参数说明
         * eq: 等于
         * ne: 不等于
         * gt: 大于
         * ge: 大于等于
         * op: 小于
         * le: 小于等于
         * cn: 包含
         * bw: 以xx开始
         * ew: 以xx结束
         */
        data: 'xxx'
    }]
}
